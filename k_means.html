1. K-Means implementation¶

K-Means is an algorithm that takes aims at partitioning a given dataset into $k$ groups (called clusters) of data similar to each other. Each cluster is characterized by its centroid. In this practice, you are going to implement the K-Means algorithm and apply it on different datasets.

Read and understand the functions below. The should_stop, get_labels and get_centroids will be implemented afterwards.

# Apply the k-means training algorithm
def kmeans(dataset, num_clusters, max_iterations=5):
    centroids = init_random_centroids(dataset, num_clusters)
    old_centroids = None
    iterations = 0
    while not should_stop(old_centroids, centroids, iterations, max_iterations):
        old_centroids = centroids
        iterations += 1
        labels = get_labels(dataset, centroids)
        centroids = get_centroids(dataset, labels, num_clusters)
    
    return centroids, labels

# Initialize centroids by choosing randomly k points from the dataset
def init_random_centroids(dataset, num_clusters):
    num_points = dataset.shape[0]
    random_indices = np.random.randint(num_points, size=num_clusters)
    random_centroids = dataset[random_indices, :]
    return random_centroids

# Returns True or False if k-means if the maximum number of iterations is reached 
# or if the centroids do not change anymore (i.e. if the distance with the previous ones is very small)
def should_stop(old_centroids, centroids, iterations, max_iterations):
    if(iterations==max_iterations): return True
    if(old_centroids is None) : return False
    return np.allclose(old_centroids, centroids)

def distance(centroid, point):
    d=0
    for i in range(centroid.shape[0]):
        d+=(centroid[i]-point[i])**2
    return np.sqrt(d)

# Returns the label for each point in the dataset in a NumPy array. The label is the index of the closest centroid
def get_labels(dataset, centroids):
    labels = np.empty(len(dataset))
    for i in range(len(dataset)):
        ind = 0
        d = distance(centroids[0],dataset[i])
        for j in range(len(centroids)):
            if(d>distance(centroids[j],dataset[i])):
                ind=j
                d=distance(centroids[j],dataset[i])
        labels[i] = ind           
    return labels

# Returns the centroids of the clusters.
# Each centroid is the geometric mean of the points that have that centroid's label.
# If a centroid is empty (no points have this centroid's label) you should randomly re-initialize it.

def get_centroids(dataset, labels, k):
    geo_mean = np.zeros([k,dataset.shape[1]])
    for i in range(k):
        bool_idx = (labels==i) # labels [1,2,1,3] -> pour labels==1 : bool_idx = [T,F,T,F]
        select = dataset[bool_idx] # garde uniquement les valeurs de dataset pour les index True
        if(select.size!=0): geo_mean[i] = np.mean(select, axis = 0)
    return geo_mean
